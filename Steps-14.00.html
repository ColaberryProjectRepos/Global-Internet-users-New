Creating Measuring/Dax table 14
Step 8: Creating Measuring/Dax table :
  1. On the home page click on the transform data
  2. Duplicate the main table (Global Internet Users). Then rename it to the Measures/Dax table.
  3. in the Measures/Dax table Right-click on the ID column and remove other columns.
Follow the following GIF to create a Measuring/Dax table
  • After creating the Measure table right click on it and create the Following Measures (Dax Formula)Total Internet users =  Sum ('Global Internet Users'[No. of Internet Users])
    1. Internet users with No filter effect = CALCULATE([Total Internet users], Calendar[Decades],Calendar[Date].[year])
    2. Cellular Subscription YTD = (AVERAGE('Global Internet Users'[Cellular Subscription]))
    3. Broadband Subscription YTD =
      (AVERAGE('Global Internet Users'[Broadband Subscription])) 
  1. BroadbandSubscriptionCY = SUM('Global Internet Users'[Broadband Subscription])
  2. CellularSubscriptionCY = SUM('Global Internet Users'[Cellular Subscription])
  3. BroadbandSubscriptionCY = SUM('Global Internet Users'[Broadband Subscription])
  4. BroadbandSubscriptionLY = CALCULATE([BroadbandSubscriptionCY], PREVIOUSYEAR('Calendar'[Date]))
  5. CellularSubscriptionLY = CALCULATE([CellularSubscriptionCY], PREVIOUSYEAR('Calendar'[Date]))
  6. Internet users LY = CALCULATE([internet users CY], PREVIOUSYEAR(Calendar[Date]))
  7. InterenetUser Indicator Arrow =
               VAR PositiveIcon = UNICHAR(9650)
               VAR NegativeIcon = UNICHAR(9660) 
               VAR Result = IF([internet users CY] > [internet users LY], PositiveIcon, NegativeIcon) 
     Return Result

  8. BroadbandSubscription Indicator Arrow =
               VAR PositiveIcon = UNICHAR(9650)
               VAR NegativeIcon = UNICHAR(9660) 
               VAR Result = IF([BroadbandSubscriptionCY] > [BroadbandSubscriptionLY], PositiveIcon, NegativeIcon)
     Return Result
  9. CellularSubscription Indicator Arrow =
               VAR PositiveIcon = UNICHAR(9650)
               VAR NegativeIcon = UNICHAR(9660) 
               VAR BlackIcon    = UNICHAR(9650)
               VAR Result = IF([CellularSubscriptionCY] > [CellularSubscriptionLY], PositiveIcon, NegativeIcon)
     Return Result 

  10. internetUsers Indicator Arrow color = IF('Measuring/Dax'[internet users CY] > 'Measuring/Dax'[internet users lY] , "#00BF00" , "red") 
  11. CellularSubscription Indicator Arrow color = IF('Measuring/Dax'[CellularSubscriptionCY] > 'Measuring/Dax'[CellularSubscriptionLY] , "#00BF00" , "red") 
  12. BroadbandSubscription Indicator Arrow color = IF('Measuring/Dax'[BroadbandSubscriptionCY] > 'Measuring/Dax'[BroadbandSubscriptionLY] , "#00BF00" , "red") 
  13. Internet users YoY% = 
    VAR CurrentYearInternetusers = [Total Internet users]
    VAR PreviousYearInternetusers = CALCULATE([Total Internet users] , DATEADD('Calendar'[Date], -1, YEAR))
    RETURN
    DIVIDE(CurrentYearInternetusers - PreviousYearInternetusers, PreviousYearInternetusers) 
  14. Cellular %Yoy =
    VAR __PREV_YEAR = CALCULATE(
        SUM(
            'Global Internet Users'[Cellular Subscription]
        ),
        DATEADD(
            'Calendar'[Date],
            -1,
            YEAR
        )
    )
    RETURN
        DIVIDE(
            SUM('Global Internet Users'[Cellular Subscription]) - __PREV_YEAR, __PREV_YEAR,0)
  15. Broadband %Yoy = VAR __PREV_YEAR = CALCULATE(
        SUM(
            'Global Internet Users'[Broadband Subscription]
        ),
        DATEADD(
            'Calendar'[Date].[Date],
            -1,
            YEAR
        )
    )
    RETURN
        DIVIDE(
            SUM('Global Internet Users'[Broadband Subscription]) - __PREV_YEAR, __PREV_YEAR,0)
  16. BroadbandSubscription %CY vs LY =
    VAR BroadbandSubscriptionCY = SUM('Global Internet Users'[Broadband Subscription])
    VAR BroadbandSubscriptionLY = CALCULATE(BroadbandSubscriptionCY,PREVIOUSYEAR('Calendar'[Date]))

    Return
        DIVIDE(BroadbandSubscriptionCY -    BroadbandSubscriptionLY,BroadbandSubscriptionLY)
  17. CellularSubscription %CY vs LY =
    VAR CellularSubscriptionCY = SUM('Global Internet Users'[Cellular Subscription])
    VAR CellularSubscriptionLY = CALCULATE(CellularSubscriptionCY,PREVIOUSYEAR('Calendar'[Date]))
    Return
    DIVIDE(CellularSubscriptionCY - CellularSubscriptionLY,CellularSubscriptionLY)
  18. InternetUsers %CY vs LY =
    Var InternetUsersCY = SUM('Global Internet Users'[No. of Internet Users])
    Var InternetUsersLY = CALCULATE(InternetUsersCY, PREVIOUSYEAR('Calendar'[Date]))
     Return
     DIVIDE( InternetUsersCY - InternetUsersLY,InternetUsersLY)
  19. Broadband Sub VS Cellular sub % = [Broadband Subscription YTD] / [Cellular Subscription YTD]
